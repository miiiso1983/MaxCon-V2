طبقة التخصيص العراقية

نظام تخصيص شامل للسوق العراقي في MaxCon ERP

نظرة عامة

تم تطوير طبقة تخصيص شاملة للسوق العراقي تتضمن دعماً كاملاً للغة العربية، والدينار العراقي كعملة أساسية، والنظام الضريبي العراقي، وقوالب الفواتير الحكومية، وتنسيقات الهاتف العراقية، والبنوك المحلية، بالإضافة إلى تقارير جاهزة للجهات الرسمية.

خطوات التفعيل
1

تفعيل مقدم الخدمة

تم تسجيل مقدم الخدمة تلقائياً في bootstrap/providers.php

2

نشر ملفات التكوين

php artisan vendor:publish --tag=iraq-config
3

نشر ملفات الترجمة

php artisan vendor:publish --tag=iraq-lang
4

نشر قوالب العرض

php artisan vendor:publish --tag=iraq-views
5

تحديث متغيرات البيئة

IRAQ_LOCALIZATION_ENABLED=true
APP_LOCALE=ar
APP_TIMEZONE=Asia/Baghdad
أمثلة الاستخدام

تنسيق العملة:

// تنسيق المبلغ بالدينار العراقي
format_iraqi_currency(1500000); // "1,500,000 د.ع"

// تحويل إلى كلمات
iraqi_currency_to_words(1500000); // "مليون وخمسمائة ألف دينار عراقي"

التحقق من أرقام الهواتف:

// التحقق من رقم الهاتف
validate_iraqi_phone('07801234567'); // true

// تنسيق الرقم
format_iraqi_phone('07801234567'); // "+964 780 123 4567"

// اكتشاف الشركة
detect_iraqi_operator('07801234567'); // ['name_ar' => 'زين العراق', ...]

حساب الضرائب:

// حساب ضريبة الدخل
calculate_iraqi_income_tax(2000000);

// حساب ضريبة الشركات
calculate_iraqi_corporate_tax(5000000, ['expenses' => 1000000]);
